type error =
  1. | Fetch_error of Jx.t
    (*

    An error occurred during fetch request.

    *)
  2. | Unsuccessful of Stdweb.Fetch.Response.t
    (*

    A non-200 response from the server.

    *)
  3. | Decoding_error of exn
    (*

    Error when decoding payload.

    *)
  4. | Handling_error of exn
    (*

    Error when handling successful response payload.

    *)
